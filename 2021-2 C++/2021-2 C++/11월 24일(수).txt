	std::cout << "賃 葆葬檣陛? ";

		if (uid(dre))

			continue;
		}

		animals[i] = new Bird(std::to_string(i + 1) + "廓 億");
	}

	for (int i = 0; i < num; ++i)
	{
		animals[i]->Move();
	}

	delete[] animals;

	Save(std::vector<std::string>{ "main.cpp", "Animal.h", "Animal.cpp" }, "11錯 24橾(熱).txt");
}
